git

config

git config --global user.name "John Cornellier"
git config --global user.email john@johnsdocs.com
git config --global alias.a add
git config --global alias.co commit
git config --global alias.st status

clone a project

git clone https://github.com/cornellier/johnsdocs.git
cd johnsdocs

create and add a file

echo .DS_Store >> .gitignore
git add .gitignore

commit

git commit .gitignore -m "add .DS_Store"

diff with origin

git fetch origin
git diff origin -- [local-path]

undo add before commit

git reset foo.txt

ammend the last commit message

git commit --amend

undo the last commit

git commit -m "oops"
git reset HEAD~

compare local with remote

git fetch
git log -p HEAD..FETCH_HEAD

remove local uncommited changes

tracked unstaged: modified but not added: git checkout foo.txt

tracked staged: have been added, aka moved to staging aka added to index: git reset --hard

untracked: new files, by definition unstaged, remove (no undo): git clean -f. To remove directories add -d.

push to remote repo

git push
